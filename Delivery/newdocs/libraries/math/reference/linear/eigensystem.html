class interface
     EIGENSYSTEM

create

     make (matrix: BASIC_MATRIX)
                 -- Accept 'matrix' as system matrix
                 -- making it ready for eigenvalues
                 -- and eigenvectors queries,
                 -- including the generalized versions of those.
                 --
                 -- Note: Nag does not support complex, non-symmetric
                 -- systems (yet).
           require
                 matrix_not_void: matrix /= void
                 matrix_quadratic: matrix.quadratic
                 matrix_real_or_blended_complex: matrix.real or matrix.blended_complex
                 matrix_symmetric_when_complex: matrix.complex implies matrix.symmetric
           ensure
                 system_set: system = matrix
                 default_tolerance_set: tolerance < 0.0
                 iterations_exists: iterations /= void

feature -- Access

     extra: BASIC_MATRIX
                 -- If not user defined then this is a square unit matrix.
                 -- Otherwise this is the stored extra matrix B
                 -- appearing in the equation A|x(i)> = a(i)*B|x(i)>.
                 -- The a(i) is the i-th generalized eigenvalue
                 -- and |x(i)> is the i-th generalized eigenvector.
           ensure
                 result_not_void: Result /= void
                 result_same_kind_as_system: Result.same_kind (system)
                 result_same_size_as_system: Result.height = system.height
                 result_quadratic: Result.quadratic

     system: BASIC_MATRIX
                 -- Quadratic system matrix A appearing in
                 -- the eigensystem equation:
                 -- A |x(i)> = a(i) |x(i)>
                 -- or its generalized version:
                 -- A |x(i)> = a(i) B |x(i)>.
     
feature -- Status report

     initial_symmetricity: BOOLEAN
                 -- To prevent an incidental
                 -- change of system symmetricity
                 -- this is initially acquired
                 -- and enforced by invariant clause.

     iterations: ARRAY [INTEGER]
                 -- Number of iterations that was
                 -- needed for each eigenvalue
                 -- as reported by Nag
                 -- (for real, unsymmetric systems)

     nag_failed: BOOLEAN
                 -- Did the last call to a NAG routine fail?
                 -- (from NAG_EXCEPTIONS)

     nag_ignoring_errors: BOOLEAN
                 -- Will nag_check raise exception if nag_failed?
                 -- (from NAG_EXCEPTIONS)

     nag_printing: BOOLEAN
                 -- Will error messages be printed by NAG?
                 -- (from NAG_EXCEPTIONS)

     nag_recovery: BOOLEAN
                 -- Will NAG return control to us?
                 -- (from NAG_EXCEPTIONS)

     tolerance: DOUBLE
                 -- If negative, then the machine
                 -- precision will be used.
                 -- Used when computing generalized
                 -- eigenvalues and eigenvectors
                 -- of real, unsymmetric systems
     
feature -- Status setting

     nag_disable_error_printing
                 -- Disable printing of error messages by NAG.
                 -- The error message will be in the argument to raise if recovering.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)

                 nag_will_not_print_errors: not nag_printing

     nag_disable_error_recovery
                 -- Tell NAG to abort if there is an error.
                 -- (from NAG_EXCEPTIONS)

                 will_abort: not nag_recovery

     nag_do_not_ignore_errors
                 -- nag_check will raise exception if nag_failed.
                 -- (from NAG_EXCEPTIONS)

                 not_nag_ignoring_errors: not nag_ignoring_errors

     nag_enable_error_printing
                 -- Set NAG to print an error message itself if an error occurs.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_will_print_errors: nag_printing

     nag_enable_error_recovery
                 -- Tell NAG to return in case of an error.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 will_return: nag_recovery

     nag_ignore_errors
                 -- nag_check will not raise exception.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_ignoring_errors: nag_ignoring_errors

                 extra_real: extra.real
                 extra_quadratic: extra.quadratic
                 extra_good_size: extra.height = system.height

feature -- Basic operations

     eigenvalues: BASIC_VECTOR
                 -- Vector of eigenvalues a(i);
                 -- each one satisfying A|x(i)> = a(i)|x(i)>,
                 -- where |x(i)> is i-th eigenvector.
                 --
                 -- This feature calls the Nag library once only;
                 -- repeated invocations of this feature will respond
                 -- with the vector already stored after the first call.
                 -- For full documentation see NAG Library routines:
                 -- - f02aac - for real, symmetric systems
                 -- - f02afc - for real, non-symmetric systems
                 -- - f02awc - for complex, symmetric systems
                 -- (complex, unsymmetric systems are not supported by Nag)
           require
                 not_complex_unsymmetric: system.complex implies system.symmetric
           ensure
                 result_not_void: Result /= void
                 result_real_when_system_symmetric: system.symmetric implies Result.real
                 result_complex_when_system_unsymmetric: not system.symmetric implies Result.complex
                 result_good_count: Result.count = system.height

     eigenvectors: BASIC_MATRIX
                 -- Matrix of eigenvectors |x(i)> stored column_wise;
                 -- each one satisfying eqs: A|x(i)> = a(i)|x(i)>,
                 -- where a(i) is i-th corresponding eigenvalue.
                 --
                 -- This feature calls the Nag library once only;
                 -- repeated invocations of this feature will respond
                 -- with the matrix already stored after the first call,
                 --
                 -- (This feature also upgrades the eigenvalues
                 -- if a Nag call is issued)
                 -- For full documentation see NAG Library routines:
                 -- - f02abc - for real, symmetric systems
                 -- - f02agc - for real, non-symmetric systems
                 -- - f02axc - for complex, symmetric systems
                 -- (complex unsymmetric systems are not supported by Nag)
           require
                 not_complex_unsymmetric: system.complex implies system.symmetric
           ensure
                 result_not_void: Result /= void
                 result_real_when_system_real_symmetric: (system.real and system.symmetric) implies Result.real
                 result_complex_otherwise: not (system.real and system.symmetric) implies Result.complex
                 result_good_count: Result.height = system.height
                 result_quadratic: Result.quadratic

     generalized_eigenvalues: BASIC_VECTOR
                 -- Vector of generalized eigenvalues a(i);
                 -- each one satisfying A|x(i)> = a(i)B|x(i)>,
                 -- where |x(i)> is i-th generalized eigenvector,
                 -- A is system matrix and B is extra matrix.
                 --
                 -- This feature calls the Nag library once only
                 -- for each new extra matrix defined;
                 -- repeated invocations of this feature
                 -- will respond with the matrix
                 -- already stored after the first call.
                 --
                 -- For full documentation see NAG Library routines:
                 -- - f02adc - for real, symmetric system and positive_definite extra
                 -- - f02bjc - for any real system and any real extra
                 -- - (complex systems are not supported by Nag)
           require
                 system_real: system.real
                 extra_real: extra.real
           ensure
                 result_not_void: Result /= void
                 result_real_case: system.symmetric and extra.positive_definite implies Result.real
                 result_complex_case: not system.symmetric or not extra.positive_definite implies Result.complex
                 result_good_size: Result.count = system.height

     generalized_eigenvectors: BASIC_MATRIX
                 -- Matrix of generalized eigenvectors |x(i)>
                 -- -- stored column-wise;
                 -- each one satisfying the equation:
                 -- A|x(i)> = a(i)B|x(i)>,
                 -- where a(i) is i-th generalized eigenvalue,
                 -- A is system matrix and B is extra matrix.
                 --
                 -- This feature calls the Nag library once only
                 -- for each new extra matrix defined;
                 -- repeated invocations of this feature
                 -- will respond with the matrix
                 -- already stored after the first call.
                 --
                 -- (This feature also upgrades the eigenvalues
                 -- if a Nag call is issued)
                 -- For full documentation see NAG Library routines:
                 -- - f02aec - for real symmetric system
                 -- - and real, positive definite extra matrix
                 -- - f02bjc - for any real systems and any real extras
                 -- (complex systems are not supported by Nag)
           require
                 system_real: system.real
                 extra_real: extra.real
           ensure
                 result_not_void: Result /= void
                 result_real_case: system.symmetric and extra.positive_definite implies Result.real
                 result_complex_case: not system.symmetric or not extra.positive_definite implies Result.complex
                 result_good_size: Result.height = system.height
                 result_quadratic: Result.quadratic

           require -- from NAG_EXCEPTIONS
