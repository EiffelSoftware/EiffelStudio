class interface
     STANDARD_ONE_STEP_RUNGE_KUTTA_ODE_SOLVER

create

     make (x0, x1: DOUBLE; y0: like y)
                 -- Initialize problem for integrating from x0 to x1
                 -- with initial value y0 (x0).
                 -- Must be used as a creation procedure.
                 -- (from ODE_SOLVER)
           require -- from ODE_SOLVER
                 y0_exists: y0 /= void
                 y0_not_empty: not y0.is_empty
           ensure -- from ODE_SOLVER
                 first_x: x = x0 and xstart = x0
                 final_x: xfinal = x1
                 y_set: equal (y0, y)
                 y_not_empty: not y.is_empty
                 tolerance_set: tolerance = default_tolerance

feature -- Access

     number_of_equations: INTEGER

                 -- (from ODE_SOLVER)

     x: DOUBLE
                 -- The independent variable.
                 -- (from ODE_SOLVER)

     xfinal: DOUBLE
                 -- Ultimate stopping value of the independent variable.
                 -- (from ODE_SOLVER)

     xstart: DOUBLE
                 -- Initial value of the independent variable.
                 -- (from ODE_SOLVER)

     y: ARRAY [DOUBLE]
                 -- The dependent variable, the solution at x.
                 -- (from ODE_SOLVER)

                 -- (from RUNGE_KUTTA_ODE_SOLVER)

     next_step_size: DOUBLE
                 -- Planned next step size. (hnext in Nag document)
                 -- (from RUNGE_KUTTA_ODE_SOLVER)
           require -- from RUNGE_KUTTA_ODE_SOLVER
                 started: started

                 -- Has the integration commenced?
                 -- (from RUNGE_KUTTA_ODE_SOLVER)

     wasted_steps: DOUBLE
                 -- Fraction of failed steps. (waste in Nag document)

                 nwant_large_enough: nwant >= 1
                 started: started

                 nwant_components: Result.count = nwant

     interp_y (xwant: DOUBLE; nwant: INTEGER): ARRAY [DOUBLE]
                 -- In one step method, get y(1..nwant) at xwant after passing xwant.
                 -- (from ONE_STEP_RUNGE_KUTTA_ODE_SOLVER)
           require -- from ONE_STEP_RUNGE_KUTTA_ODE_SOLVER
                 not_with_order_8: method_order <= 5
                 nwant_small_enough: nwant <= y.count
                 nwant_large_enough: nwant >= 1
                 started: started
           ensure -- from ONE_STEP_RUNGE_KUTTA_ODE_SOLVER
                 nwant_components: Result.count = nwant

     output (xx: DOUBLE; yy: ARRAY [DOUBLE])
                 -- Invoke output in target class.

                 -- (from ONE_STEP_RUNGE_KUTTA_ODE_SOLVER)

           ensure -- from ODE_SOLVER
