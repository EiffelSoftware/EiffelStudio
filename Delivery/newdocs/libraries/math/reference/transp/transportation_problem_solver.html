class interface
     TRANSPORTATION_PROBLEM_SOLVER

create

     make
                 -- Create the solver.

feature -- Access

     available: ARRAY [DOUBLE]
                 -- Goods available at each source

     cost: BASIC_MATRIX
                 -- Cost of moving goods from source i to destination j

     maximum_iterations: INTEGER
                 -- Maximum number of iterations allowed.

     required: ARRAY [DOUBLE]
                 -- Goods required at each destination
     
feature -- Status report

     consistent_input: BOOLEAN
                 -- Are the inputs consistent and proper?
           require
                 requiredd_exists: required /= void
                 available_exists: available /= void
                 cost_exists: cost /= void
           ensure
                 correct_height: Result implies cost.height = number_of_sources
                 correct_width: Result implies cost.width = number_of_destinations
                 real_cost: Result implies cost.real
                 balanced: Result implies effectively_equal (sum_total (required), sum_total (available))

     Default_maximum_iterations: INTEGER is 200

     destination_numbers: ARRAY [INTEGER]
                 -- Destination indices for the optimal solution

     number_of_destinations: INTEGER
                 -- The number of destinations
           require
                 requirements: required /= void
           ensure
                 Result = available.count

     number_of_iterations: INTEGER
                 -- Number of iterations performed by last solve

     number_of_sources: INTEGER
                 -- The number of source locations
           require
                 availabilities: available /= void
           ensure
                 Result = available.count

     optimal_cost: DOUBLE
                 -- Cost of the optimal solution

     optimal_quantities: ARRAY [DOUBLE]
                 -- Array of optimal quantities found by solve.

     solution_size: INTEGER
                 -- Common size of the solution arrays
                 -- Available after solve.

     source_numbers: ARRAY [INTEGER]
                 -- Source indices for the optimal solution

     unit_costs: ARRAY [DOUBLE]
                 -- Unit costs associated with route for source/destination
     
feature -- Element change

     set_available (a: ARRAY [DOUBLE])
                 -- Set available to a.
           require
                 not_void: a /= void
                 a_long_enough: a.count > 0
           ensure
                 available_set: available = a

     set_cost (a: BASIC_MATRIX)
                 -- Set cost to a.
           require
                 not_void: a /= void
                 a_real: a.real
                 a_wide_enough: a.width >= 1
                 a_tall_enough: a.height >= 1
           ensure
                 cost_set: cost = a
                 cost_real: cost.real

     set_maximum_iterations (m: INTEGER)
                 -- Set maximum_iterations to m.
           require
                 m_large_enough: m > 0
           ensure
                 maximum_iterations_set: maximum_iterations = m

     set_required (a: ARRAY [DOUBLE])
                 -- Set required to a.
           require
                 not_void: a /= void
                 a_long_enough: a.count > 0
           ensure
                 required_set: required = a
     
feature -- Basic operations

     effectively_equal (a, b: DOUBLE): BOOLEAN
                 -- Are a and b equal to a relative error
                 -- <= machine precision
           require
                 a_large_enough_abs: a.abs >= smallest_number
           ensure
                 Result implies (((a - b) / a).abs <= machine_precision)

     solve
                 -- Solve the transportation problem.
           require
                 consistent_input: consistent_input
           ensure
                 optimal_quantities_exist: optimal_quantities /= void
                 source_numbers_exist: source_numbers /= void
                 destination_numbers_exist: destination_numbers /= void
                 unit_costs_exist: unit_costs /= void
                 source_numbers_count: source_numbers.count = solution_size
                 destination_numbers_count: destination_numbers.count = solution_size
                 unit_costs_count: unit_costs.count = solution_size
                 optimal_quantities_count: optimal_quantities.count = solution_size

     sum_total (a: ARRAY [DOUBLE]): DOUBLE
                 -- Sum of the array's entries
           require
                 not_void: a /= void
     
feature -- Machine parameters

     machine_precision: DOUBLE
                 -- Smallest positive number such that 1 + x /= x
                 -- (from MATHEMATICIAN)

     smallest_number: DOUBLE
                 -- Smallest positive number that can be inverted.
                 -- (from MATHEMATICIAN)
     
invariant

     maximum_iterations_positive: maximum_iterations > 0
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class TRANSPORTATION_PROBLEM_SOLVER
