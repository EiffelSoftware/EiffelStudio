class interface
     NONLINEAR_SYSTEM_SOLVER

create

     make (t: like target)
                 -- Create a solver for transform t.
                 -- Set default tolerances.
           require
                 t_not_void: t /= void
                 t_square: t.n = t.m

feature -- Access

     tolerance: DOUBLE
                 -- Accuracy required in solution.
     
feature -- Status report

     check_jacobian (x: ARRAY [DOUBLE])
                 -- Test the relation between jacobian and value at `x.
           require
                 using_jacobian: using_jacobian
                 x_not_void: x /= void
                 right_size: x.count = target.n

     do_not_use_jacobian
                 -- Do not use the Jacobian function in solving system.
           ensure
                 no_jacobian: not using_jacobian

     nag_failed: BOOLEAN
                 -- Did the last call to a NAG routine fail?
                 -- (from NAG_EXCEPTIONS)

     nag_ignoring_errors: BOOLEAN
                 -- Will nag_check raise exception if nag_failed?
                 -- (from NAG_EXCEPTIONS)

     nag_printing: BOOLEAN
                 -- Will error messages be printed by NAG?
                 -- (from NAG_EXCEPTIONS)

     nag_recovery: BOOLEAN
                 -- Will NAG return control to us?
                 -- (from NAG_EXCEPTIONS)

     set_defaults
                 -- Sets suggested tolerance from NAG manual c05nbc.
                 -- Uses Jacobian if possible.

     use_jacobian
                 -- Use Jacobian if possible (default) in solving system.
           ensure
                 yes_if_possible: target.has_jacobian implies using_jacobian

     using_jacobian: BOOLEAN
                 -- Is the Jacobian to be used?
     
feature -- Status setting

     nag_disable_error_printing
                 -- Disable printing of error messages by NAG.
                 -- The error message will be in the argument to raise if recovering.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_will_not_print_errors: not nag_printing

     nag_disable_error_recovery
                 -- Tell NAG to abort if there is an error.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 will_abort: not nag_recovery

     nag_do_not_ignore_errors
                 -- nag_check will raise exception if nag_failed.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 not_nag_ignoring_errors: not nag_ignoring_errors

     nag_enable_error_printing
                 -- Set NAG to print an error message itself if an error occurs.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_will_print_errors: nag_printing

     nag_enable_error_recovery
                 -- Tell NAG to return in case of an error.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 will_return: nag_recovery

     nag_ignore_errors
                 -- nag_check will not raise exception.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_ignoring_errors: nag_ignoring_errors
     
feature -- Element change

     set_tolerance (t: DOUBLE)
                 -- Set tolerance to t.
           require
                 positive: t >= 0.0
           ensure
                 tolerance_set: tolerance = t
     
feature -- Conversion

     to_nag: POINTER
                 -- An instance of C type NagError, for passing to NAG routines.
                 -- (from NAG_EXCEPTIONS)
     
feature -- Basic operations

     nag_check
                 -- Check result of calling a NAG routine.
                 -- Raise a user exception if it failed.
                 -- (from NAG_EXCEPTIONS)

     nag_error_history: NAG_ERROR_HISTORY
                 -- History of error messages from NAG.
                 -- (from NAG_EXCEPTIONS)

     solve (guess: ARRAY [DOUBLE])
                 -- Find the root, result to last_root.
           require -- from SYSTEM_ROOT_FINDER
                 target_set: target /= void
                 target_square: target.n = target.m
                 guess_supplied: guess /= void
                 guess_suitable: guess.count = target.n
           ensure -- from SYSTEM_ROOT_FINDER
                 correct_size: last_root.count = target.n
     
feature -- Failure information

     nag_last_error_aux: INTEGER
                 -- Last auxillary error information from NAG call.
                 -- (from NAG_EXCEPTIONS)
           require -- from NAG_EXCEPTIONS
                 nag_failed

     nag_last_error_code: INTEGER
                 -- Last NAG error code from failed NAG call.
                 -- (from NAG_EXCEPTIONS)
           require -- from NAG_EXCEPTIONS
                 nag_failed

     nag_last_error_message: STRING
                 -- Error message from failed NAG call.
                 -- (from NAG_EXCEPTIONS)
           require -- from NAG_EXCEPTIONS
                 nag_failed
     
feature -- Parameters

     last_root: ARRAY [DOUBLE]
                 -- Result of last call to solve.
                 -- (from SYSTEM_ROOT_FINDER)

     last_value: ARRAY [DOUBLE]
                 -- `target.value (last_root)'.
                 -- (from SYSTEM_ROOT_FINDER)

     target: TRANSFORMATION
                 -- The system to be solved.
                 -- (from SYSTEM_ROOT_FINDER)
     
feature -- Parameters change

     set_target (t: like target)
                 -- Set target to t.
                 -- (from SYSTEM_ROOT_FINDER)
           ensure -- from SYSTEM_ROOT_FINDER
                 target_set: target = t
