class interface
     NAG_ERROR_HISTORY

create

     make
                 -- Initialize a new empty history.

feature -- Access

     error (n: INTEGER): NAG_ERROR_MESSAGE
                 -- Copy of the n-th last error.
           require
                 not_too_small: n > 0
                 not_too_big: n <= error_count

     history: TWO_WAY_LIST [NAG_ERROR_MESSAGE]
                 -- The current list of remembered messages, oldest first.
     
feature -- Measurement

     Default_error_capacity: INTEGER is 10
                 -- Default value of error_capacity.

     error_capacity: INTEGER
                 -- Maximum number of errors to remember.
                 -- Made irrelevant by remember_all_errors.

     error_count: INTEGER
                 -- Number of errors available in this history.
           ensure
                 equals_history_count: Result = history.count

     error_total: INTEGER
                 -- Number of errors so far, remembered or not.

     warning_total: INTEGER
                 -- Number of the errors so far, remembered or not,
                 -- which were warnings.
     
feature -- Status report

     errors_recorded: BOOLEAN
                 -- Are any errors recorded?
           ensure
                 result_equals_count_positive: Result = (error_count > 0)

     fatal_recorded: BOOLEAN
                 -- Did a non-warning error ever occur?

     unbounded_capacity: BOOLEAN
                 -- Is the maximum capacity free to increase by itself?
     
feature -- Conversion

     out: STRING
                 -- Printable form of the list.
     
feature -- Modification

     put (nr: NAG_ERROR_MESSAGE)
                 -- Record a new error.
           require
                 nr_exists: nr /= void
           ensure
                 remembered: (error_capacity > 0) implies ((error_count > 0) and (nr = history.i_th (error_count)))

     remember_all_errors
                 -- Increase error_capacity as needed.
           ensure
                 unbounded_capacity: unbounded_capacity

     remember_errors (n: INTEGER)
                 -- Remember up to n errors.
           require
                 n_non_negative: n >= 0
           ensure
                 bounded: not unbounded_capacity
                 correct_capacity: error_capacity = n
                 trimmed: history.count <= error_capacity
     
invariant

     history_not_void: history /= void
     non_negative_error_count: error_count >= 0
     non_negative_capacity: error_capacity >= 0
     correct_capacity: error_capacity >= error_count
     non_negative_total: error_total >= 0
     correct_total: error_total >= error_count
     capacity_rule: unbounded_capacity implies error_capacity > 0
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class NAG_ERROR_HISTORY
