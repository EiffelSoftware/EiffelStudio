     packed_complex: BOOLEAN
                 -- True if Current is complex and if its components
                 -- have half_complex symmetry and are stored in one array.
                 --
                 -- Other names: half complex, hermitian, conjugate symmetric.
                 -- (Avoid using the last two aliases because they have other
                 -- meaning in standard matrix text books).
                 --
                 -- Packed complex matrix is kind of complex requiring only half
                 -- of the storage due to specific symmetry imposed on each row.
                 -- This symmetry allows for a recovery of the full information
                 -- and is being heavily used in section C06 of Nag library
                 -- (Fourier transforms).
                 --
                 -- If 'a' stands for real, and 'b' - for imaginary, then:
                 -- For even number (n = 2s) of columns the storage order in
                 -- each row is:
                 -- a(1), a(2),..a(s), a(s+1), b(s), b(s-1)..b2
                 --
                 -- For odd number of columns (n = 2s - 1):
                 -- a(1), a(2)..a(s),b(s),b(s-1),..b(2) - with a(s+1)=0
                 --
                 -- b(1) is always zero, a(s+1) is zero for odd sequences

     positive_definite: BOOLEAN
                 -- Is Current positive definite?

     silent_dimension (a, b: BASIC_MATRIX; ma, mb: INTEGER): INTEGER
                 -- Silent dimension defined as follows:
                 -- When two matrices A, B enter a product
                 -- operation A * B in a straightforward fashion
                 -- the silent dimension is equal to the width
                 -- of A and the height of B.
                 --
                 -- When any of those matrices will be transposed
                 -- or conjugate transpose during such an operation
                 -- - as in the feature 'compound'
                 -- their dimensions must still match properly.
                 --
                 -- This feature answers the value of such
                 -- a dimension or zero if there is a dimension
                 -- mismatch among matrices A, B, C,
                 -- to be used in the operation:
                 -- C := alpha * A' * B' + beta * C,
                 -- (' signifies possible transposition),
                 -- where C is Current matrix, and
                 -- alpha and beta are scalars.
                 --
                 -- See 'compound'
                 -- for details on markers ma and mb
           require
                 matrix_a_exists: a /= void
                 matrix_b_exists: b /= void
                 good_marker_a: ma >= 0 and ma < 3
                 good_marker_b: mb >= 0 and mb < 3
           ensure
                 result_non_negative: Result >= 0

     set_symmetricity (status: BOOLEAN)
                 -- Providing that Current is quadratic...
                 -- One of the triangles will be ignored
                 -- by Nag Library if quadratic matrix
                 -- is declared symmetric by you, e.g.
                 -- the following will be assumed:
                 -- a(j,i) = a (i,j) for real matrices
                 -- and a(j,i) = a(i,j) conjugated
                 -- for complex matrices.
                 -- You need too consult a specific section
                 -- of Nag manual to learn which triangle
                 -- will be ignored by a given call.
                 -- Default of feature 'symmetric' is False.
                 --
                 -- Note: Complex symmetric matrices are
                 -- usually called 'hermitian'.
           require
                 is_quadratic: quadratic

     compound (a, b: BASIC_MATRIX; alpha, beta: COMPLEX; ma, mb: INTEGER)
                 --
                 -- Perform compound operation:
                 -- C := alpha * A' * B' + beta * C
                 -- where C is Current object,
                 -- A, B are two other matrices,
                 -- alpha and beta are complex scalars.
                 --
                 -- Markers ma and mb signify
                 -- how the matrices A and B will
                 -- enter such an operation:
                 -- 0 - no transposition (e.g., A' = A)
                 -- 1 - transposition (e.g., A' = A transposed)
                 -- 2 - conjugation and transposition
                 -- (e.g., A' = A conjugate_transposed)
                 -- (same as 1 for real matrices)
                 --
                 -- If Current is real then A and B must be real
                 -- and imaginary parts of alpha and beta will be ignored.
                 -- If Current is blended_complex then A and B must be
                 -- blended_complex as well.
                 --
                 -- For details see Nag library procedures:
                 -- - f06yac - for real matrices

                 -- |y> := A | x>, and
                 -- <y| := <x| A,

                 -- <x|, <y| are row vectors.
