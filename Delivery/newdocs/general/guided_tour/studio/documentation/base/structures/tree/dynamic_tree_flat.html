							orgidx2_stack.put (p2.child_index)

							Result := False
						end

						Result := False
					end

							p2 := t2_stack.item

							t2_stack.remove
							idx_stack.remove

							orgidx2_stack.remove
						end
					end
				end
				if not Result then

					invariant

					until
						orgidx1_stack.count = 1

						orgidx2_stack.remove
					end
