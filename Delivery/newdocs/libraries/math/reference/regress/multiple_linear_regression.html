class interface
     MULTIPLE_LINEAR_REGRESSION

create

     make (x: SELECTABLE_DATA; y: ARRAY [DOUBLE]; around_zero: BOOLEAN)
                 -- Find linear regression for y over x.
                 -- If around_zero, result is to have constant term = 0.0.
           require
                 x_not_void: x /= void
                 y_not_void: y /= void
                 same_count: x.number_of_observations = y.count
                 number_of_observations_large_enough: x.number_of_observations >= 2
                 number_of_selected_variables_small_enough: x.number_of_selected_variables <= x.number_of_selected_observations
                 number_of_selected_variables_small_enough_including_mean: around_zero or (x.number_of_selected_variables <= (x.number_of_selected_observations - 1))
           ensure
                 number_of_observations_correct: number_of_observations = x.number_of_selected_observations
                 number_of_independent_variables_correct: (around_zero and (number_of_independent_variables = x.number_of_selected_variables)) or ((not around_zero) and (number_of_independent_variables = x.number_of_selected_variables + 1))

     make_with_tolerance (x: SELECTABLE_DATA; y: ARRAY [DOUBLE]; around_zero: BOOLEAN; tolerance: DOUBLE)
                 -- Find linear regression for y over x.
                 -- If around_zero, result is to have constant term = 0.0.
           require
                 x_not_void: x /= void
                 y_not_void: y /= void
                 same_count: x.number_of_observations = y.count
                 number_of_observations_large_enough: x.number_of_selected_observations >= 2
                 number_of_variables_large_enough: x.number_of_selected_variables >= 1
                 number_of_selected_variables_small_enough: x.number_of_selected_variables <= x.number_of_selected_observations
                 number_of_selected_variables_small_enough_including_mean: around_zero or (x.number_of_selected_variables <= (x.number_of_selected_observations - 1))
                 tolerance_large_enough: tolerance >= 0.0
           ensure
                 number_of_observations_correct: number_of_observations = x.number_of_selected_observations
                 number_of_independent_variables_correct: (around_zero and (number_of_independent_variables = x.number_of_selected_variables)) or ((not around_zero) and (number_of_independent_variables = x.number_of_selected_variables + 1))
