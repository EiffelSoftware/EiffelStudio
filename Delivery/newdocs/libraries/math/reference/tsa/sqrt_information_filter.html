class interface
     SQRT_INFORMATION_FILTER

feature -- Access

     a_inverse: BASIC_MATRIX
                 -- Inverse of the state transition matrix

     b: BASIC_MATRIX
                 -- Model input weight matrix B (or BA^-1 )

     c: BASIC_MATRIX
                 -- Model output weight matrix C (not an output of iterate) or C R^(-1/2)

     q_inverse: BASIC_MATRIX
                 -- Right Cholesky factor of the process noise covariance matrix Q^(-1/2)

     r_inverse: BASIC_MATRIX
                 -- Right Cholesky factor of the input of the measurement noise covariance matrix.

     r_inverse_y: BASIC_VECTOR
                 -- R^(-1/2) * Y, where Y is the measured output

     t: BASIC_MATRIX
                 -- S(i), Cholesky form, modified by iterate.

     x: BASIC_VECTOR
                 -- Estimated state (modified by iterate).

     z: BASIC_VECTOR
                 -- The mean value of the state process noise.
     
feature -- Measurement

     m: INTEGER
                 -- Width of b
           require
                 b_set: b /= void
           ensure
                 width_of_b: Result = b.width

     n: INTEGER
                 -- Size of t (and a)
           require
                 t_set: t /= void
           ensure
                 size_of_t: Result = t.height and Result = t.width

     p: INTEGER
                 -- Number of rows of c
           require
                 c_set: c /= void
           ensure
                 size_of_c: (Result = c.height)

     u: BASIC_MATRIX
                 -- The unitary transformation calculated by apply_unitary_transform
     
feature -- Status report

     b_separate: BOOLEAN
                 -- Is b representing B as opposed to A^(-1) B?

     input_set: BOOLEAN
                 -- True if all required inputs are set
           ensure
                 inputs_all_set: Result implies ((t /= void) and (a_inverse /= void) and (q_inverse /= void) and (c /= void) and (b /= void) and ((r_inverse /= void) or (not rc_separate)) and (x /= void) and (r_inverse_y /= void) and (z /= void))

     is_time_invariant: BOOLEAN
                 -- Is this a time invariant problem, input set as per g13ebc?

     rc_separate: BOOLEAN
                 -- Are R^(-1/2) and C being input separately or is c = CR^(-1/2)

     tolerance: DOUBLE
                 -- Input tolerance for testing of near singularity of h
     
feature -- Status setting

     set_a_inverse (new_a: BASIC_MATRIX)
                 -- Change the value of a_inverse
           require
                 new_a_exists: new_a /= void
                 new_a_real: new_a.real
                 quadradic: new_a.height = new_a.width
           ensure
                 a_set: a_inverse = new_a
                 quadradic: a_inverse.height = a_inverse.width

     set_b (new_b: BASIC_MATRIX; is_separate: BOOLEAN)
                 -- Set b to new_b, indicating if b represents just B or A^(-1) B.
           require
                 new_b_exists: new_b /= void
                 new_b_real: new_b.real
           ensure
                 b_set: b = new_b
                 b_separate_set: b_separate = is_separate

     set_c (new_c: BASIC_MATRIX)
                 -- Set the value of c representing C R^(-1/2).
           require
                 new_c_exists: new_c /= void
                 new_c_real: new_c.real
           ensure
                 c_set: c = new_c

                 rc_not_separate: not rc_separate

     set_q_inverse (new_q: BASIC_MATRIX)

                 new_ry_exists: new_ry /= void
                 new_ry_real: new_ry.real

                 new_r_real: new_r.real

                 new_c_real: new_c.real
                 quadratic_r: new_r.height = new_r.width
                 correct_size: new_c.height = new_r.height
           ensure
                 rc_separate: rc_separate
                 r_inverse_set: r_inverse = new_r
                 c_set: c = new_c
                 r_inverse_real: r_inverse.real
                 c_real: c.real
                 correct_size: r_inverse.height = c.height

     set_t (new_t: BASIC_MATRIX)
                 -- Change the value of t
           require
                 new_t_exists: new_t /= void
                 new_t_real: new_t.real
                 quadradic: new_t.height = new_t.width
           ensure
                 t_set: t = new_t
                 quadradic: t.height = t.width

           ensure
