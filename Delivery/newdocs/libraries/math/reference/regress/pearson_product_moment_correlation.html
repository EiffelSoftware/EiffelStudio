class interface
     PEARSON_PRODUCT_MOMENT_CORRELATION

create

     make (x: SELECTABLE_DATA)
                 -- Create Pearson product moment correlations for set x.
           require
                 x_not_void: x /= void
                 enough_observations: x.number_of_selected_observations >= 1
                 enough_variables: x.number_of_selected_variables >= 1
           ensure
                 observation_count: number_of_observations <= x.number_of_selected_observations
                 weight_check: x.weighted or (number_of_observations = x.number_of_selected_observations)
                 variable_count: number_of_variables = x.number_of_selected_variables

feature -- Initialization

     make (x: SELECTABLE_DATA)
                 -- Create Pearson product moment correlations for set x.
           require
                 x_not_void: x /= void
                 enough_observations: x.number_of_selected_observations >= 1
                 enough_variables: x.number_of_selected_variables >= 1
           ensure
                 observation_count: number_of_observations <= x.number_of_selected_observations
                 weight_check: x.weighted or (number_of_observations = x.number_of_selected_observations)
                 variable_count: number_of_variables = x.number_of_selected_variables
     
feature -- Access

     correlation_coefficients: BASIC_MATRIX
                 -- Matrix containing set of correlation coefficients.

     covariance: BASIC_MATRIX
                 -- The variance-covariance matrix

     means: BASIC_VECTOR
                 -- The means of the observations for each variable.

     selected_data: SELECTABLE_DATA
                 -- The data actually used in the correlation_coefficients

     standard_deviations: BASIC_VECTOR
                 -- The standard deviations of the observations for each variable.

     sum_of_weights: DOUBLE
                 -- The sum of weights of the observations.

     weights: BASIC_VECTOR
                 -- The weights of each observation
           ensure
                 result_exists: Result /= void
                 result_real: Result.real
                 result_count: Result.count = number_of_observations
     
feature -- Measurement

     number_of_observations: INTEGER
                 -- Number of observations used in the correlation_coefficients

     number_of_variables: INTEGER
                 -- Number of variables used in the correlation_coefficients
     
invariant

     enough_variables: number_of_variables >= 1
     enough_observations: number_of_observations >= 1
     selected_data_exists: selected_data /= void
     data_width: selected_data.data.width = number_of_variables
     data_real: selected_data.data.real
     data_height: selected_data.data.height = number_of_observations
     correlation_coefficients_exists: correlation_coefficients /= void
     covariance_exists: covariance /= void
     correlation_coefficients_real: correlation_coefficients.real
     covariance_real: covariance.real
     correlation_coefficients_width: correlation_coefficients.width = number_of_variables
     correlation_coefficients_height: correlation_coefficients.height = number_of_variables
     covariance_width: covariance.width = number_of_variables
     covariance_height: covariance.height = number_of_variables
     means_exist: means /= void
     means_real: means.real
     means_count: means.count = number_of_variables
     standard_deviations_exist: standard_deviations /= void
     standard_deviations_real: standard_deviations.real
     standard_deviations_count: standard_deviations.count = number_of_variables
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)
