     status: "See notice at end of class."
     date: "$Date$"
     revision: "$Revision$"

class interface
     EIFMATH_SYMMETRIC_MATRIX

create

     make (n: INTEGER)
                 -- Make of dimension n.
           require
                 positive_size: n > 0
           ensure
                 dimension_set: dimension = n

feature -- Access

     item (i, j: INTEGER): DOUBLE
                 -- The item at `(i, j)'.
           require
                 valid_index_i: valid_index (i)
                 valid_index_j: valid_index (j)
     
feature -- Measurement

     dimension: INTEGER
                 -- The order of the matrix.

     valid_index (i: INTEGER): BOOLEAN
                 -- Is i a valid index in either position?
           ensure
                 in_range: Result implies (i >= 1 and i <= dimension)
     
feature -- Element change

     put (v: DOUBLE; i, j: INTEGER)
                 -- Make item at `(i, j)' equal to v.
           require
                 valid_index_i: valid_index (i)
                 valid_index_j: valid_index (j)
           ensure
                 right_item: item (i, j) = v
                 symmetric: item (j, i) = v
     
feature -- Conversion

     to_array: ARRAY [DOUBLE]
                 -- The matrix as an array of contiguous doubles.
     
invariant

     to_array_not_void: to_array /= void
     positive_dimension: dimension > 0
     square: to_array.count = dimension * dimension
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class EIFMATH_SYMMETRIC_MATRIX
