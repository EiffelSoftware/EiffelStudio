feature -- Basic operations

     integrate
                 -- Integrate the function using d01ajc from the NAG library,
                 -- using a general adaptive method.
           require -- from INTEGRATOR
                 integrand_ready: integrand_ready
                 method_ready: method_ready
           ensure then -- from NAG_ADAPTIVE_INTEGRATOR

     integrate_alg_weighted (alpha, beta: DOUBLE; choice: INTEGER)
                 -- Integrate the function using d01apc from the NAG library.
                 -- Choice selects weighting function:
                 -- choice 1: (x-a)^alpha * (b-x)^beta
                 -- choice 2: (x-a)^alpha * (b-x)^beta * ln(x-a)
                 -- choice 3: (x-a)^alpha * (b-x)^beta * ln(x-b)
                 -- choice 4: (x-a)^alpha * (b-x)^beta * ln(x-a) * ln(x-b)
           require
                 integrand_ready: integrand_ready
                 method_ready: method_ready
                 maximum_large_enough: maximum_subintervals >= 2
           ensure
                 integrated: integrated

     integrate_cauchy_weighted (c: DOUBLE)
                 -- Integrate the function using d01aqc from the NAG library.
           require
                 integrand_ready: integrand_ready
                 method_ready: method_ready
           ensure
                 integrated: integrated

     integrate_cosine_weighted (omega: DOUBLE)
                 -- Integrate the function using d01anc from the NAG library.
           require
                 integrand_ready: integrand_ready

           ensure
                 integrated: integrated
