class interface
     ORTHOGONAL_TRANSFORMER_MINIMIZER

create

     make (matrix: BASIC_MATRIX)
                 -- Adopt matrix as system matrix
                 -- in preparation for seeking
                 -- solutions of Least Squares problems.
                 -- Decompose the system matrix
                 -- into Q*R factors, where Q is:
                 -- unitary for complex systems or
                 -- orthogonal for real system.
                 --
                 -- Matrix R is upper triangular - real or complex.
                 -- For details see Nag Library functions:
                 -- - f01qcc - for real matrices
                 -- - f01rcc - for complex matrices
           require
                 matrix_not_void: matrix /= void
                 matrix_real_or_blended_complex: matrix.real or matrix.blended_complex
                 matrix_not_wide: matrix.height >= matrix.width
           ensure
                 system_set: system = matrix
                 factors_not_void: factors /= void
                 factors_same_kind: factors.same_kind (system)
                 factors_height_correct: factors.height = system.height
                 factors_width_correct: factors.width = system.width
                 zeta_not_void: zeta /= void
                 zeta_same_kind: zeta.same_kind (system)
                 zeta_count_correct: zeta.count = system.width

feature -- Access

     factors: BASIC_MATRIX
                 -- Details of Q*R factorization
                 -- Upper triangle is R - real or complex.
                 -- Bottom portion of 'factors' and vector 'zeta'
                 -- keep data for the reconstruction of
                 -- of matrix 'Q' - orthogonal_transformer.
                 -- For details see Nag Library functions:
                 -- - f01qcc - for real matrices
                 -- - f01rcc - for complex matrices

     orthogonal_transformer: BASIC_MATRIX
                 -- Quadratic symetric matrix Q
                 -- given explicite -
                 -- instead of its internal representation
           ensure
                 result_exists: Result /= void
                 result_quadratic: Result.quadratic
                 result_count_correct: Result.height = system.height
                 result_same_kind: Result.same_kind (system)

     orthogonal_transformer_columns (n: INTEGER): BASIC_MATRIX
                 -- Matrix with explicit values of
                 -- 'n' columns of the transformer Q
                 -- (Result may be wider than requested)
                 -- For details see Nag Library functions:
                 -- - f01qec - for real systems
                 -- - f01rec - for complex systems
           require
                 not_too_few_columns: n > 0
                 not_too_many_columns: n <= system.height
           ensure
                 result_not_void: Result /= void
                 result_width_not_too_small: Result.width >= n
                 result_width_not_too_large: Result.width <= system.height

     system: BASIC_MATRIX
                 -- Original, not factorized system matrix

     zeta: BASIC_VECTOR
                 -- Auxiliary array
                 -- used to reconstruct the transformer Q.
                 -- See also: 'factors

feature -- Transformation

     transform (matrix: BASIC_MATRIX; transposed_q: BOOLEAN)
                 -- Transform 'matrix' by performing Q * 'matrix'
                 -- if 'transposed_Q' is False,
                 -- or by performing (Q conjugated transpose) * 'matrix'
                 -- if 'transposed_Q' is True.
                 -- (Congugate transpose = transpose for real matrices)
                 -- For details see Nag Library functions:
                 -- - f01qdc - for real matrices
                 -- - f01rdc - for complex matrices
           require
                 matrix_not_void: matrix /= void
                 matrix_height_correct: matrix.height = system.height
                 matrix_same_kind: matrix.same_kind (system)
