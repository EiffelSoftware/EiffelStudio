indexing
	description: "Comparators for single characters"
	status: "See note at end of class"
	date: "$Date$"
	revision: "$Revision$"

class interface
	SINGLE_CHARACTER_COMPARATOR

create 

	make (c: CHARACTER)
			-- Create comparator.
		ensure
			character_set: character = c

feature -- Access

	character_set: STRING
			-- Character represented by comparator

	contains (c: CHARACTER): BOOLEAN
			-- Does comparator contain c?
	
invariant

		-- from ANY
	reflexive_equality: standard_is_equal (Current)
	reflexive_conformance: conforms_to (Current)
		-- from COMPARATOR
	no_empty_comparator: character_set /= void and then not character_set.is_empty

end -- class SINGLE_CHARACTER_COMPARATOR
