feature -- Access

	access_date: INTEGER
			-- Date of last access

	buffered_file_info: SPECIAL [CHARACTER]
			-- Special data zone
			-- (from TO_SPECIAL)

	change_date: INTEGER
			-- Date of last status change

	date: INTEGER
			-- Last modification date

	device: INTEGER
			-- Device number on which inode resides

	device_type: INTEGER
			-- Device type on which inode resides

	file_name: STRING
			-- File name to which information applies.

	group_id: INTEGER
			-- GID of the file

	group_name: STRING
			-- Name of the file group, if available from /etc/group.
			-- Otherwise, the GID

	inode: INTEGER
			-- Inode number

	links: INTEGER
			-- Number of links

	owner_name: STRING
			-- Name of the file owner, if available from /etc/passwd.
			-- Otherwise, the UID

	protection: INTEGER
			-- Protection mode of file (12 lower bits)

	size: INTEGER
			-- File size, in bytes

	type: INTEGER
			-- File type (4 bits, 12 lowest bits zeroed)

	user_id: INTEGER
			-- UID of the file owner
