create

     make (matrix: BASIC_MATRIX)
                 -- Accept matrix as system of rows to operate on.
                 -- A particular case is a row BASIC_VECTOR,
                 -- one-row basic matrix of minimum count 2.
                 -- System matrix is a shallow copy of 'matrix',
                 -- and any of the transformations will be stored
                 -- back in the original 'matrix'.
           require
                 matrix_not_void: matrix /= void
                 matrix_wide_enough: matrix.width > 1
           ensure
                 system_exists: system /= void
                 system_same_width: system.width = matrix.width
                 system_same_height: system.height = matrix.height
                 system_not_blended: not system.blended_complex

feature -- Transformation

     transform_using_cosine_fft (forward: BOOLEAN)
                 -- Do Cosine transformation from real matrix
                 -- to real matrix. Since Cosine transform is its own
                 -- inverse it does not matter whether 'forward'
                 -- is True or False;
                 -- Interface to Nag library function c06hbc
           require
                 system_is_real: system.real
                 system_wide_enough: system.width > 1
           ensure
                 system_is_real: system.real

     transform_using_quarter_cosine_fft (forward: BOOLEAN)
                 -- Do quarter-wave Fourier Cosine transform
                 -- from real matrix to real matrix.
                 -- To restore the original system call this
                 -- feature again with reversed value of 'forward'.
                 -- Interface to Nag Library function c06hdc
           require
                 system_is_real: system.real
                 system_wide_enough: system.width > 1
           ensure
                 system_is_real: system.real

     transform_using_quarter_sine_fft (forward: BOOLEAN)
                 -- Do quarter-wave Fourier Sine transform
                 -- from real matrix to real matrix.
                 -- To restore the original system call this
                 -- feature again with reversed value of 'forward'.
                 -- Interface to Nag Library function c06hcc
           require
                 system_is_real: system.real
                 system_wide_enough: system.width > 1
           ensure
                 system_is_real: system.real

     transform_using_sine_fft (forward: BOOLEAN)
                 -- Do Sine transformation from real matrix
                 -- to real matrix. Since Sine transform is its own
                 -- inverse it does not matter whether 'forward'
                 -- is True or False.
                 -- Interface to Nag library function c06hac.
           require
                 system_is_real: system.real
                 system_wide_enough: system.width > 1
           ensure
                 system_is_real: system.real

     transform_using_standard_fft (forward: BOOLEAN)
                 -- Do FFT transformation of the system matrix.
                 -- You can use the same call again but with
                 -- reversed value of 'forward' to restore
                 -- the original data of the system matrix.
                 --
                 -- Six cases are handled (transparently to user)
                 -- by this feature:
                 -- -- Single transformation of one row matrices
                 -- -- (or row-vectors):
                 -- 1. Real to packed_complex - See Nag Library function c06eac
                 -- 2. Packed_complex to real - See Nag Library function c06ebc
                 -- 3. Dual_complex to dual_complex - See Nag Library function c06ecc
                 -- -- Multiple transformations - system matrix height > 1:
                 -- 4. Real to packed_complex - See Nag Library function c06fpc
                 -- 5. Packed_complex to real - See Nag Library function c06fqc
                 -- 6. Dual_complex to dual_complex - See Nag Library function c06frc
                 --
                 -- See also notes re packed_complex
                 -- inside class BASIC_MATRIX.
           require
                 system_not_blended: not system.blended_complex
                 system_wide_enough: system.width > 1
           ensure
                 system_not_blended: not system.blended_complex

     transform_using_two_dimensional_fft (forward: BOOLEAN)
                 -- Do two-dimensional discrete Fourier transform
                 -- of a bivariate sequence of complex data values
                 -- stored in system matrix.
                 -- To restore the original system repeat this
                 -- call with a reversed value of 'forward'.
                 -- Interface to Nag procedure c06fuc.
           require
                 system_is_dual_complex: system.dual_complex
                 system_wide_enough: system.width > 1
           ensure
                 system_is_dual_complex: system.dual_complex

feature -- Implementation

     eifmath_c06eac (x_count: INTEGER; x_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of real sequence.
                 -- Interface to Nag library procedure c06eac.
                 -- (from NAG_FOURIER)

     eifmath_c06ebc (x_count: INTEGER; x_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of hermitian sequence.
                 -- Interface to Nag library procedure c06ec.
                 -- (from NAG_FOURIER)

     eifmath_c06ecc (x_count: INTEGER; x_area: SPECIAL [DOUBLE]; y_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of complex data
                 -- stored in parallel:
                 -- x_area = real, y_area = imaginary part.
                 -- Interface to Nag library procedure c06ecc.
                 -- (from NAG_FOURIER)

     eifmath_c06ekc (convolute: BOOLEAN; x_count: INTEGER; x_area: SPECIAL [DOUBLE]; y_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Convolution (convolute = True)
                 -- or correlation (convolute = False)
                 -- of two real sequences: x_area and y_area.
                 -- Interface to Nag library procedure c06ekc.
                 -- (from NAG_FOURIER)

     eifmath_c06fpc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of m real sequences
                 -- stored sequentially in x_area
                 -- (as in matrix, row-wise).
                 -- Array trig_area has count 2*n
                 -- and contains trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06fpc.
                 -- (from NAG_FOURIER)

     eifmath_c06fqc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of m hermitian sequences
                 -- stored sequentially in x_area
                 -- (as in matrix, row-wise).
                 -- Array trig_area has count 2*n
                 -- and contains trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06fqc.
                 -- (from NAG_FOURIER)

     eifmath_c06frc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; y_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- FFT of m complex sequences
                 -- stored sequentially in x_area (real)
                 -- and in y_area (imaginary)
                 -- (as in matrix, row-wise).
                 -- Array trig_area has count 2*n
                 -- and contains trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06frc
                 -- (from NAG_FOURIER)

     eifmath_c06fuc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; y_area: SPECIAL [DOUBLE]; trig_m_area: SPECIAL [DOUBLE]; trig_n_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Two dimensional FFT of bivariate sequence
                 -- of complex data - with real parts
                 -- in x_area and imaginary parts in y_area.
                 -- The size of both arrays is m*n
                 -- and the order of storage is as
                 -- in matrix - row-wise.
                 -- Arrays trig_m_area and trig_n_area
                 -- have count 2*m and 2*n, respectively,
                 -- and contain trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06fuc.
                 -- (from NAG_FOURIER)

     eifmath_c06gbc (x_count: INTEGER; x_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Complex conjugation of hermitian sequence.
                 -- Interface to Nag library procedure c06gbc.
                 -- (from NAG_FOURIER)

     eifmath_c06gcc (x_count: INTEGER; x_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Complex conjugation of imaginary sequence
                 -- (just a negation).
                 -- Interface to Nag library procedure c06gcc.
                 -- (from NAG_FOURIER)

     eifmath_c06gqc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Complex conjugation of m hermitiani sequences
                 -- Count of x is m*n.
                 -- Interface to Nag library procedure c06gqc.
                 -- (from NAG_FOURIER)

     eifmath_c06gsc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; u_area: SPECIAL [DOUBLE]; v_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Convertion of m hermitian sequences
                 -- stored serially in x_area (as in matrix m*n)
                 -- to m complex sequences stored sequentially
                 -- in u_area (real) and v_area (imaginary).
                 -- Interface to Nag library procedure c06gsc.
                 -- (from NAG_FOURIER)

     eifmath_c06gzc (n: INTEGER; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Trigonometrical coefficients
                 -- required for the computations
                 -- of FFT's for multiple sequences.
                 -- Count of x = 2*n.
                 -- Interface to Nag library procedure c06gzc.
                 -- (from NAG_FOURIER)

     eifmath_c06hac (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Sine transform of m real sequences
                 -- of count 'n - 1' each.
                 -- Trig_area has 2*n trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06hac.
                 -- (from NAG_FOURIER)

     eifmath_c06hbc (m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Cosine transform of m real sequences
                 -- of count 'n + 1' each.
                 -- Trig_area has 2*n trigonometrical
                 -- coefficients.
                 -- Interface to Nag library procedure c06hbc.
                 -- (from NAG_FOURIER)

     eifmath_c06hcc (forward: BOOLEAN; m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Quater-wave sine transform of m real sequences
                 -- of count n each.
                 -- Trig_area has 2*n trigonometrical
                 -- coefficients.
                 -- If forward = True then forward transform
                 -- is performed, otherwise the transform
                 -- is done backward (inverse).
                 -- Interface to Nag library procedure c06hcc.
                 -- (from NAG_FOURIER)

     eifmath_c06hdc (forward: BOOLEAN; m: INTEGER; n: INTEGER; x_area: SPECIAL [DOUBLE]; trig_area: SPECIAL [DOUBLE]; fail: POINTER)
                 -- Quater-wave cosine transform of m real sequences
                 -- of count n each.
                 -- Trig_area has 2*n trigonometrical
                 -- coefficients.
                 -- If forward = True then forward transform
                 -- is performed, otherwise the transform
                 -- is done backward (inverse).
                 -- Interface to Nag library procedure c06hdc.
                 -- (from NAG_FOURIER)
