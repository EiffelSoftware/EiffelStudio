create

     make (r: DOUBLE)
                 -- Make a bivariate standard normal
                 -- distribution with correlation coefficient r.
           require
                 r_large_enough: r >= - 1.0
                 r_small_enough: r <= 1.0
           ensure
                 rho_set: rho = r
                 rho_proper: (r <= 1.0) and (r >= - 1.0)
