	description: "Comparator for character sets"
	status: "See note at end of class"
	date: "$Date$"
	revision: "$Revision$"
	usage: "The features 'define', 'add' and 'remove' accept a string asargument that represents a set of single character (like 'a')or character ranges (like 'a-z'). To use a character with aspecial meaning (like '-') as single character, it can be quotedusing the character '\'"

class interface
	CHARACTER_SET

feature -- Access

	comparator_builder: COMPARATOR_BUILDER_IMPL
			-- Comparator builder singleton
			-- (from COMPARATOR_BUILDER)

	contains_character (c: CHARACTER): BOOLEAN
			-- Does character set contain character c?
		require
			not_empty: not is_empty
		ensure
			index_unchanged: comparators.index = old comparators.index

	contains_string (s: STRING): BOOLEAN
			-- Does character set contain string s?
		require
			not_empty: not is_empty
			string_exists: s /= void

	has (token: STRING): BOOLEAN
			-- Is there a comparator for token?
		require
			non_empty_token: token /= void and then not token.is_empty
		ensure
			index_unchanged: comparators.index = old comparators.index

	set: STRING
			-- Current character set
		require
			not_empty: not is_empty
		ensure
			index_unchanged: comparators.index = old comparators.index
	
feature -- Status report

	is_empty: BOOLEAN
			-- Is character set empty?
	
feature -- Status setting

	define (s: STRING)
			-- Define character set.
		require
			string_exists: s /= void
		ensure
			comparators_set_up: comparators /= void and then not comparators.is_empty
	
feature -- Element change

	add (s: STRING)
			-- Add s to character set.
		require
			non_empty_string: s /= void and then not s.is_empty
			no_such_comparator: not has (s)
		ensure
			elements_added: comparators.count >= old comparators.count
	
feature -- Removal

	remove (s: STRING)
			-- Remove s from character set.
		require
			non_empty_string: s /= void and then not s.is_empty
			has_comparator: has (s)
		ensure
			elements_removed: comparators.count <= old comparators.count
	
invariant

		-- from ANY
	reflexive_equality: standard_is_equal (Current)
	reflexive_conformance: conforms_to (Current)

end -- class CHARACTER_SET
