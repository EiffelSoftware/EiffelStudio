class interface
     EIFFEL_FEATURE

feature -- Access

     class_name: STRING
                 -- The name of the class to which the feature belongs

     f: POINTER
                 -- Pointer to the feature

     feature_name: STRING
                 -- Name of the feature or other description.

     target: ANY
                 -- Target on which to call the feature
     
feature -- Status report

     ready: BOOLEAN
                 -- Are the components all set in a consistent way?
                 -- Note that there is no way currently to be certain
                 -- that target conforms to the class of f but it must.
           ensure
                 Result implies (target /= void and f /= default_pointer)
     
feature -- Element change

     set (t: like target; f_name: STRING)
                 -- Make feature point to feature f_name in object t.
           require
                 t_not_void: t /= void
                 f_name_not_void: f_name /= void
           ensure
                 target_set: target = t
                 f_set: f /= default_pointer

     set_by_name (c_name: STRING; f_name: STRING)
                 -- Point to feature f_name in c_name.
           require
                 c_name_not_void: c_name /= void
                 f_name_not_void: f_name /= void
           ensure
                 f_set: f /= default_pointer

     set_by_pointer (t: like target; f_name: STRING; p: POINTER)
                 -- Point to feature whose address is p in class of t.
                 -- f_name used to set "feature_name" but need not be
                 -- the real name of the feature.
           require
                 p_not_default_pointer: p /= default_pointer
                 t_not_void: t /= void
                 f_name_exists: f_name /= void
           ensure
                 f_set: f /= default_pointer
                 target_set: t = target

     set_target (t: ANY)
                 -- Set target to t.
           ensure
                 target_set: target = t
     
invariant

           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class EIFFEL_FEATURE
