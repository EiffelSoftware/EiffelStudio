     solve (a: BASIC_MATRIX; bl, bu, cvec, x: BASIC_VECTOR): DOUBLE
                 -- Objective value of linear programming problem with general linear constraints
                 -- Supplying a is optional, use Void if not desired.
                 -- Supply cvec for linear programming problem, Void for feasible point problem
           require

                 x_long_enough: x.count > 0
                 bl_real: bl.real
                 bu_real: bu.real
                 a_real: (a = void) or else a.real
                 a_size: (a = void) or else x.count = a.width
                 cvec_real: (cvec = void) or else cvec.real
                 x_cvec_size: (cvec = void) or else x.count = cvec.count
                 bl_size_no_a: (a = void) implies bl.count = x.count
                 bl_size_with_a: (a /= void) implies bl.count = x.count + a.height
                 bu_size_no_a: (a = void) implies bu.count = x.count
                 bu_size_with_a: (a /= void) implies bu.count = x.count + a.height
