create

     make_repeatable (x, vars: DOUBLE; c, a: ARRAY [DOUBLE]; seed: INTEGER)
                 -- Make an ARMA series with mean x, autoregressive coefficients c,
                 -- and moving average coefficients m, variance of the errors vars.
                 -- c and/or 'a' may be Void if desired.
                 -- Use seed to intialize the random number generator.
           require
                 vars_positive: vars > 0
           ensure
                 mean_set: mean = x
                 variance_set: variance = vars

                 theta_set: equal (theta, a)

     make_non_repeatable (x, vars: DOUBLE; c, a: ARRAY [DOUBLE])
                 -- Make an ARMA series with mean x, autoregressive coefficients c,
                 -- and moving average coefficients m, variance of the errors vars.

           require
