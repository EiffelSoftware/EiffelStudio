feature -- Status setting

     set_final (state, f: INTEGER) is
                 -- Make state a final state for regular expression f.
           require
                 is_in_automaton: state <= count and state >= 1
           local
                 old_p: CURSOR
           do
                 old_p := cursor
                 go_i_th (state)
                 item.set_final (f)
                 go_to (old_p)
           ensure
                 same_index: index = old index
           end
