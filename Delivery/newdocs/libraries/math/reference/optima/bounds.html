feature -- Access

     Bound_huge_negative: DOUBLE is -1.0e10

     Bound_huge_positive: DOUBLE is 1.0e10

     lower_bounds: ARRAY [DOUBLE]
                 -- Array of lower bounds for the solution
                 -- (abscissas)

     upper_bounds: ARRAY [DOUBLE]
                 -- Array of upper bounds for the solution
     
feature -- Measurement

     bounds_count: INTEGER
                 -- Number of bounds
     
feature -- Comparison

     bounds_consistent (lower, upper: ARRAY [DOUBLE]): BOOLEAN
                 -- Are user supplied bounds consistent?
           require
                 lower_not_void: lower /= void
                 upper_not_void: upper /= void
                 lower_bounds_count_correct: lower.count = bounds_count
                 upper_bounds_count_correct: upper.count = bounds_count
     
feature -- Status report

     bounds_type: INTEGER
                 -- Type of bounds, enumerated
                 -- by Nag Library.

     bounds_type_verbose: STRING
                 -- Verbose version of type of bounds.
                 --
                 -- These are the string aliases
                 -- to the Nag enumerated symbols
                 -- (shown in parenthesis) appearing
                 -- in section E04 of Nag Library:
                 --
                 -- "none" (Nag_NoBounds),
                 -- "distinct" (Nag_Bounds),
                 -- "zero" (Nag_BoundsZero),
                 -- "equal" (Nag_BoundsEqual)
           ensure
                 result_exists: Result /= void
                 result_unique: Result.is_equal ("none") or Result.is_equal ("zero") or Result.is_equal ("distinct") or Result.is_equal ("equal")
     
feature -- Status setting

     set_bounds_distinct (lower, upper: ARRAY [DOUBLE])
                 -- Set all lower bounds to 'lower'
                 -- Set all upper bounds to 'upper'
           require
                 lower_not_void: lower /= void
                 upper_not_void: upper /= void
                 lower_bounds_count_correct: lower.count = bounds_count
                 upper_bounds_count_correct: upper.count = bounds_count
                 bounds_consistent: bounds_consistent (lower, upper)
           ensure
                 bounds_distinct_set: bounds_type_verbose.is_equal ("distinct")
                 lower_bounds_set: lower_bounds = lower
                 upper_bounds_set: upper_bounds = upper

     set_bounds_equal (lower, upper: DOUBLE)
                 -- Set all lower bounds to 'lower'
                 -- Set all upper bounds to 'upper'
           require
                 bounds_consistent: lower < upper
                 lower_bounds_exists: lower_bounds /= void
                 upper_bounds_exists: upper_bounds /= void
                 bounds_count_large_enough: bounds_count > 0
           ensure
                 bounds_equal_set: bounds_type_verbose.is_equal ("equal")

     set_bounds_none
                 -- Remove all the bounds
           ensure
                 bounds_none_set: bounds_type_verbose.is_equal ("none")

     set_bounds_zero
                 -- Set all lower bounds to zero
                 -- (To be explicitely done by Nag,
                 -- but we will do it here anyway)
           require
                 lower_bounds_exists: lower_bounds /= void
                 upper_bounds_exists: upper_bounds /= void
                 bounds_count_large_enough: bounds_count > 0
           ensure
                 bounds_zero_set: bounds_type_verbose.is_equal ("zero")
