class interface
     LINEAR_REGRESSION

create

     make (x: SELECTABLE_DATA; y: ARRAY [DOUBLE]; around_zero: BOOLEAN)
                 -- Find linear regression of the selected set of y's over x.
                 -- If around_zero, result is to have a constant term = 0.0.
           require
                 x_not_void: x /= void
                 y_not_void: y /= void
                 same_count: x.number_of_observations = y.count
                 count_large_enough: x.number_of_selected_observations >= 1
                 count_large_enough_for_degrees_of_freedom: around_zero or x.number_of_selected_observations >= 2
                 one_dependent_variable: x.number_of_selected_variables = 1
                 not_constant_x: not is_constant (x.selected_data.data.to_array)
                 not_constant_y: not is_constant (x.select_from (y))

     make_unweighted (x, y: ARRAY [DOUBLE]; around_zero: BOOLEAN)
                 -- Find linear regression for y over x.
                 -- If around_zero, result is to have constant term = 0.0.
           require
                 x_not_void: x /= void
                 y_not_void: y /= void
                 same_count: x.count = y.count
                 count_large_enough: x.count >= 1
                 count_large_enough_for_degrees_of_freedom: around_zero or x.count >= 2
                 not_constant_x: not is_constant (x)
                 not_constant_y: not is_constant (y)

     make_weighted (x, y, wt: ARRAY [DOUBLE]; around_zero: BOOLEAN)
                 -- Find linear regression for y over x using weights wt
                 -- If around_zero, result is to have constant term = 0.0.
           require
                 x_not_void: x /= void
                 y_not_void: y /= void
                 wt_not_void: wt /= void
                 same_count: x.count = y.count and x.count = wt.count
                 count_large_enough: number_of_zero_elements (wt) <= wt.count - 1
                 count_large_enough_for_degrees_of_freedom: around_zero or number_of_zero_elements (wt) <= wt.count - 2
                 sum_large_enough: sum_of_elements (wt) >= 1.0
                 sum_large_enough_for_degrees_of_freedom: around_zero or sum_of_elements (wt) >= 2.0
                 not_constant_x: not is_constant (x)
                 not_constant_y: not is_constant (y)
                 is_nonnegative_wt: is_nonnegative (wt)

feature -- Status report

     constant_term: DOUBLE
                 -- The a in y = b x + a.

     constant_term_error: DOUBLE
                 -- The error found in estimating constant_term.

     degrees_of_freedom: DOUBLE
                 -- The degrees of freedom associated with sum_of_squared_residuals.

     linear_term: DOUBLE
                 -- The coefficient b in y = b x + a

     linear_term_error: DOUBLE
                 -- The error found in estimating linear_term

     r_squared: DOUBLE
                 -- The coefficient of determination, R**2

     sum_of_squared_residuals: DOUBLE
                 -- The sum of squares of the residuals about the regression

     value (x_value: DOUBLE): DOUBLE
                 -- Value of the regression line at x_value
           ensure
                 definition: Result = constant_term + linear_term * x_value
