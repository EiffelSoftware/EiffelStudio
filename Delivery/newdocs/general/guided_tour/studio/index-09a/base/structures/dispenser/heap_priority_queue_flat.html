				if j < count - 1 then
					left := area.item (j)
					k := j + 1
					right := area.item (k)
					if right > left then
						j := k
						left := right
					end
				elseif j = count - 1 then
					left := area.item (j)
				else
					stop := True
				end
				if not stop then
					if left > up then
						area.put (left, i)
						i := j
					else
						stop := True
					end
				end
			end
			area.put (up, i)
		end
