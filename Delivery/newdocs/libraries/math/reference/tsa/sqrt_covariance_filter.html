     date: "$Date$"

class interface
     SQRT_COVARIANCE_FILTER

feature -- Access

     a: BASIC_MATRIX
                 -- State transition matrix

     b: BASIC_MATRIX
                 -- Model input weight matrix

     c: BASIC_MATRIX
                 -- Model output weight matrix (not an output of iterate)

     q: BASIC_MATRIX
                 -- Model input noise covariance matrix

     r: BASIC_MATRIX
                 -- Measurement noise covariance matrix, Cholesky form

     s: BASIC_MATRIX
                 -- S(i), Cholesky form, modified by iterate.
     
feature -- Measurement

     h: BASIC_MATRIX
                 -- If measuring, H ^ (1/2) (set by iterate).

     k: BASIC_MATRIX
                 -- If measuring, product of the gain matrix with a (set by iterate).

     m: INTEGER
                 -- Width of b
           require
                 b_set: b /= void
           ensure
                 width_of_b: Result = b.width

     n: INTEGER
                 -- Size of s (and a)
           require
                 s_set: s /= void
           ensure
                 size_of_s: Result = s.height and Result = s.width

     p: INTEGER
                 -- Size of r
           require
                 r_set: r /= void
           ensure
                 size_of_r: (Result = r.height) and (Result = r.width)

     u: BASIC_MATRIX
                 -- The unitary transformation calculated by apply_unitary_transform
     
feature -- Status report

     bq_separate: BOOLEAN
                 -- Are B and Q being input separately or is b = BQ^(1/2)?

     input_set: BOOLEAN
                 -- True if all required inputs are set
           ensure
                 set: Result implies ((s /= void) and (a /= void) and (r /= void) and (c /= void) and (b /= void) and ((q /= void) or (not bq_separate)))

     is_time_invariant: BOOLEAN
                 -- Is this a time invariant problem, input set as per g13ebc?

     measuring: BOOLEAN
                 -- Will iterate set h and k?

     tolerance: DOUBLE
                 -- Input tolerance for testing of near singularity of h
     
feature -- Status setting

     do_not_measure
                 -- Make iterate set h and k to Void.
           ensure
                 not_measuring: not measuring

     measure
                 -- Make iterate calculate h and k.
           ensure
                 measuring: measuring

     set_a (new_a: BASIC_MATRIX)
                 -- Change the value of a
           require
                 new_a_not_void: new_a /= void
                 new_a_real: new_a.real
                 quadradic: new_a.height = new_a.width
           ensure
                 a_set: a = new_a
                 quadradic: a.height = a.width

     set_b (new_b: BASIC_MATRIX)
                 -- Change the value of b to include noise covariance term.
                 -- q will be ignored.
           require
                 new_b_not_void: new_b /= void
                 new_b_real: new_b.real
           ensure
                 bq_unified: not bq_separate
                 b_set: b = new_b

     set_bq (new_b, new_q: BASIC_MATRIX)
                 -- Change the value of b and q if desired to input them separately.
           require
                 new_q_not_void: new_q /= void
                 new_q_real: new_q.real
                 new_b_not_void: new_b /= void
                 new_b_real: new_b.real
                 correct_size: new_q.height = new_q.width and new_b.width = new_q.height
           ensure
                 bq_separate: bq_separate
                 q_set: q = new_q
                 b_set: b = new_b
                 q_real: q.real
                 b_real: b.real
                 correct_size: q.height = q.width and b.width = q.height

     set_c (new_c: BASIC_MATRIX)
                 -- Change the value of c
           require
                 new_c_not_void: new_c /= void
                 new_c_real: new_c.real
           ensure
                 c_set: c = new_c

     set_r (new_r: BASIC_MATRIX)
                 -- Change the value of r
           require
                 new_r_not_void: new_r /= void
                 new_r_real: new_r.real
                 quadratic: new_r.height = new_r.width
           ensure
                 r_set: r = new_r
                 quadratic: r.height = r.width

     set_s (new_s: BASIC_MATRIX)
                 -- Change the value of s
           require
                 new_s_not_void: new_s /= void
                 new_s_real: new_s.real
                 quadradic: new_s.height = new_s.width
           ensure
                 s_set: s = new_s
                 quadradic: s.height = s.width

     set_tolerance (t: DOUBLE)
                 -- Set tolerance to t.
           require
                 t_non_negative: t >= 0.0
           ensure
                 tolerance_set: tolerance = t

     time_invariant
                 -- Indicate that input is in form for iterate to use time_invariant method.
                 -- (NAG g13ebc)
           ensure
                 time_invariant: is_time_invariant

     time_varying
                 -- Indicate that input is in form for iterate to use time_varying method.
                 -- (NAG g13eac)
           ensure
                 time_varying: not is_time_invariant
     
feature -- Basic operations

     iterate
                 -- Do one step of the Kalman filter
           require
                 input_set: input_set
                 b_height: b.height = n
                 c_width: c.width = n
                 c_height: c.height = p
           ensure
                 k_set: measuring implies k /= void and then (k.real and k.height = n and k.width = p)
                 h_set: measuring implies h /= void and then (h.real and h.height = p and h.width = p)
                 kh_void: (not measuring) implies (k = void and h = void)
     
invariant

     positive_tolerance: tolerance >= 0.0
     s_real: s /= void implies s.real
     a_real: a /= void implies a.real
     b_real: b /= void implies b.real
     q_real: q /= void implies q.real
     c_real: c /= void implies c.real
     r_real: r /= void implies r.real
     s_quadratic: s /= void implies (s.width = s.height)
     a_quadratic: a /= void implies (a.width = a.height)
     q_quadratic: q /= void implies (q.width = q.height)
     r_quadratic: r /= void implies (r.width = r.height)
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class SQRT_COVARIANCE_FILTER
