class interface
     SINGULAR_VALUE_DECOMPOSER_MINIMIZER

create

     make (matrix: BASIC_MATRIX)
                 -- Adopt rectangular matrix [m x n] as
                 -- system matrix A.
                 -- Decompose system matrix A as
                 -- A = Q D P, where:
                 -- Q (column_vectors) is an orthogonal
                 -- quadratic matrix [m x m],
                 -- P (row_vectors) is an orthogonal
                 -- quadratic matrix [n x n],
                 -- D is real rectangular matrix [m x n]
                 -- with all zeros with the exception
                 -- of 'diagonal' elements s(i) = d(i,i).
                 --
                 -- Note: For storage efficiency
                 -- reasons Nag substitutes the above
                 -- decomposition by more economical
                 -- versions of the factors Q D P -
                 -- as described in postconditions.
                 -- You can augment those with zeros
                 -- if you want to directly test
                 -- correctness of the formula A = Q D P.
                 -- See also comments in 'row_vectors',
                 -- 'column_vectors' and 'singular_values'.
                 --
                 -- Note: Orthogonal complex matrices
                 -- are usually called 'unitary'.
                 --
                 -- For details see Nag Library functions:
                 -- - f02wec - for real systems
                 -- - f02xec - for complex systems
           require
                 matrix_not_void: matrix /= void
                 real_or_blended_complex: matrix.real or matrix.blended_complex
           ensure
                 system_defined: system = matrix
                 row_vectors_not_void: row_vectors /= void
                 column_vectors_not_void: column_vectors /= void
                 when_system_quadratic: system.quadratic implies column_vectors.quadratic and (column_vectors.height = system.height) and row_vectors.quadratic and (row_vectors.height = system.height)
                 when_system_tall: system.height > system.width implies (column_vectors.height = system.height) and (column_vectors.width = system.width) and row_vectors.quadratic and (row_vectors.height = system.width)
                 when_system_wide: system.height < system.width implies column_vectors.quadratic and (column_vectors.height = system.width) and (row_vectors.height = system.height) and (row_vectors.width = system.width)
                 singular_values_good_size: singular_values.count = system.height.min (system.width)
                 singular_values_real: singular_values.real
                 singular_values_positive: singular_values.positive
                 singular_values_decreasing: singular_values.strictly_decreasing
                 column_vectors_same_kind: column_vectors.same_kind (system)
                 row_vectors_same_kind: row_vectors.same_kind (system)
                 mathematician_exists: mathematician /= void
                 tolerance_defined: tolerance = mathematician.machine_precision

feature -- Access

     column_vectors: BASIC_MATRIX
                 -- Full or partial set of orthogonal column_vectors,
                 -- the left part Q of the factorization A = Q D P.
                 --
                 -- If system matrix is tall then only the first
                 -- 'system.width' column_vectors are stored
                 -- - each one the size of 'system.height'.
                 -- The formula A = Q D P still holds if the remaining
                 -- vectors are formed as all-zeros columns.
                 --
                 -- Otherwise all column_vectors are stored and the matrix
                 -- is fully symmetric

     mathematician: MATHEMATICIAN
                 -- Needed to get access to
                 -- 'machine_precision'

     row_vectors: BASIC_MATRIX
                 -- Full or partial set of orthogonal row vectors,
                 -- the right part of the factorization A = Q D P.
                 --
                 -- If the system matrix is wide then only the first
                 -- 'system.height' row_vectors are stored -
                 -- each one the size of 'system.width'.
                 -- The formula A = Q D P still holds if the remaining
                 -- vectors are formed as all-zeros rows.
                 --
                 -- Otherwise all row_vectors are stored and the matrix
                 -- is fully symmetric.

     singular_values: BASIC_VECTOR
                 -- Vector of singular values of real system matrix,
                 -- e.g., the elements d(i,i) of matrix D - a middle part
                 -- of factorization A = Q D P.
                 --
                 -- Vector count is equal to smaller dimension
                 -- of system matrix.
                 --
                 -- Squares of these values are also the eigenvalues
                 -- of matrix A~ * A, where A is 'system' matrix,
                 -- and symbol ~ stands for conjugate-transpose.

     system: BASIC_MATRIX
                 -- Original, not factorized
                 -- system matrix
     
feature -- Measurement

     rank: INTEGER
                 -- Number of non-negligible
                 -- strictly decreasing
                 -- components of
                 -- 'singular_values'
           require
                 singular_values.strictly_decreasing
           ensure
                 minimum_rank: Result > 0
                 maximum_rank: Result <= system.height.min (system.width)

     nag_ignore_errors
                 -- nag_check will not raise exception.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_ignoring_errors: nag_ignoring_errors

           require
                 not_too_low: epsilon >= mathematician.machine_precision
           ensure
                 tolerance_set: tolerance = epsilon

feature -- Basic operations

     nag_check
                 -- Check result of calling a NAG routine.
                 -- Raise a user exception if it failed.
                 -- (from NAG_EXCEPTIONS)

     nag_error_history: NAG_ERROR_HISTORY
                 -- History of error messages from NAG.
                 -- (from NAG_EXCEPTIONS)

     solution (b: BASIC_VECTOR): BASIC_VECTOR
                 -- Solution |x> of Least Squares Minimization Problem:
                 -- minimize scalar product <r|r>, where
                 -- |r> = |b> - A|x> is an 'error' vector,
                 -- A is system matrix,
                 -- and |b> is arbitrary vector
                 -- of count equal to height of A.
                 --
                 -- Note: This feature is more generic
                 -- than the similar feature from
                 -- Orthogonal Transformer-Minimizer
                 -- since it can be applied to
                 -- rank deficient systems, e.g.
                 -- those with rank smaller than
                 -- min (system_height, system_width),
           require
                 b_not_void: b /= void
                 not_wide: system.height >= system.width
                 column_vector: b.is_column_vector
                 good_type: b.same_kind (system)
                 good_size: b.count = system.height
           ensure
                 column_vector: Result.is_column_vector
                 good_type: Result.same_kind (system)
                 good_size: Result.count = system.width
