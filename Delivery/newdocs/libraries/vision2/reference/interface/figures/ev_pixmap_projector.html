class interface
	EV_PIXMAP_PROJECTOR

create 

	make (a_world: EV_FIGURE_WORLD; a_pixmap: EV_PIXMAP)
			-- Create with a_world and a_pixmap.
		require
			a_world_not_void: a_world /= void
			a_pixmap_not_void: a_pixmap /= void

	make_with_buffer (a_world: EV_FIGURE_WORLD; a_buffer, a_pixmap: EV_PIXMAP)
			-- Create with a_world, a_pixmap and a_buffer.
		require
			a_world_not_void: a_world /= void
			a_pixmap_not_void: a_pixmap /= void
			a_buffer_not_void: a_buffer /= void

feature -- Access

	area: EV_DRAWABLE
			-- Area associated with widget.
			-- Void if no buffer is used.
			-- (from EV_WIDGET_PROJECTOR)

	area_x: INTEGER
			-- Coordinates of top-left corner of displayed part of drawable.
			-- (0, 0) when no buffer is used.
			-- Was declared in EV_WIDGET_PROJECTOR as synonym of area_y.
			-- (from EV_WIDGET_PROJECTOR)

	area_y: INTEGER
			-- Coordinates of top-left corner of displayed part of drawable.
			-- (0, 0) when no buffer is used.
			-- Was declared in EV_WIDGET_PROJECTOR as synonym of area_x.
			-- (from EV_WIDGET_PROJECTOR)

	drawable: EV_DRAWABLE
			-- Drawable surface (screen, drawing area or pixmap).
			-- (from EV_FIGURE_DRAWER)

	world: EV_FIGURE_WORLD
			-- Figure-world that will be projected.
			-- (from EV_PROJECTOR)
	
feature -- Status report

	buffer_used: BOOLEAN
			-- Is drawable only a buffer area for area?
			-- (from EV_WIDGET_PROJECTOR)
	
feature -- Status setting

	set_world (a_world: EV_FIGURE_WORLD)
			-- Set world to a_world.
			-- (from EV_PROJECTOR)
		require -- from EV_PROJECTOR
			a_world_exists: a_world /= void
		ensure -- from EV_PROJECTOR
			world_assigned: world = a_world
	
feature -- Element change

	change_area_position (a_x, a_y: INTEGER)
			-- area has moved to (a_x, a_y) of drawable.
			-- (from EV_WIDGET_PROJECTOR)
		require -- from EV_WIDGET_PROJECTOR
			valid_position: a_x >= 0 and a_y >= 0
	
feature -- Basic operations

	clear_device
			-- Erase entire canvas.
			-- (from EV_WIDGET_PROJECTOR)

	full_project
			-- Project entire area.
			-- (from EV_WIDGET_PROJECTOR)

	is_projecting: BOOLEAN
			-- Is a project currently being performed?
			-- Then, do not start a new one.
			-- (from EV_PROJECTION_ROUTINES)

	project
			-- Make a standard projection of world on device.
			-- (from EV_WIDGET_PROJECTOR)

	register_figure (a_figure: EV_FIGURE; a_routine: PROCEDURE [ANY, TUPLE [EV_FIGURE]])
			-- Assign a_routine for drawing of a_figure.
			-- (from EV_PROJECTION_ROUTINES)

	update_rectangle (u: EV_RECTANGLE; a_x, a_y: INTEGER)
			-- Flush u on area at (a_x, a_y).
			-- (from EV_WIDGET_PROJECTOR)
		require -- from EV_WIDGET_PROJECTOR
			buffer_used: buffer_used
	
feature -- Implementation

	default_colors: EV_STOCK_COLORS
			-- Once access to EV_STOCK_COLORS.
			-- (from EV_PROJECTION_ROUTINES)
	
feature -- Basic Operations

	draw_grid
			-- Draw grid on canvas.
			-- world.point is the origin of the grid.
			-- (from EV_FIGURE_DRAWER)
	
feature -- Element Change

	set_drawable (a_drawable: EV_DRAWABLE)
			-- Set drawable to a_drawable.
			-- (from EV_FIGURE_DRAWER)
		require -- from EV_FIGURE_DRAWER
			a_drawable_not_void: a_drawable /= void
		ensure -- from EV_FIGURE_DRAWER
			assigned: drawable = a_drawable
