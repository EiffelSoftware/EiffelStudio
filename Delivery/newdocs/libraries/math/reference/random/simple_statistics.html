class interface
     SIMPLE_STATISTICS

create

     make (x: ARRAY [DOUBLE])
                 -- Create statistics for array x.
           require
                 x_not_void: x /= void
                 x_not_empty: not x.is_empty
           ensure
                 number_ok: n = x.count and n > 0
                 all_valid: valid_observations = n

     make_with_weights (x, w: ARRAY [DOUBLE])
                 -- Create statistics for array x weighted by w.
                 -- Only a reference is kept to x and w; execute
                 -- recalculate after changing their elements.
           require
                 x_not_void: x /= void
                 x_not_empty: not x.is_empty
                 w_not_void: w /= void
                 same_size: w.count = x.count
           ensure
                 number_ok: n = x.count and n > 0
