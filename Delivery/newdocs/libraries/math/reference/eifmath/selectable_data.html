class interface
     SELECTABLE_DATA

create

     make (nobs, nvars: INTEGER)
                 -- Create a new data set of nobs observations on 'nvars' variables.
                 -- All observations are initialzed to zero.
           require
                 enough_variables: nobs >= 1
                 enough_observations: nvars >= 1
           ensure
                 correct_observations: number_of_observations = nobs
                 correct_variables: number_of_variables = nvars
                 all_observations_selected: number_of_observations = number_of_selected_observations
                 all_variables_selected: number_of_variables = number_of_selected_variables
                 not_weighted: not weighted
                 weights_set: weights /= void

     make_from_matrix (x: BASIC_MATRIX)
                 -- Create data set based on matrix x.
                 -- Note: x is remembered by reference.
           require
                 x_exists: x /= void
                 x_real: x.real
                 enough_variables: x.width >= 1
                 enough_observations: x.height >= 1
           ensure
                 data_set: data = x
                 correct_observations: number_of_observations = x.height
                 correct_variables: number_of_variables = x.width
                 all_observations_selected: number_of_observations = number_of_selected_observations
                 all_variables_selected: number_of_variables = number_of_selected_variables
                 not_weighted: not weighted
                 weights_set: weights /= void

feature -- Initialization

     make (nobs, nvars: INTEGER)
                 -- Create a new data set of nobs observations on 'nvars' variables.
                 -- All observations are initialzed to zero.
           require
                 enough_variables: nobs >= 1
                 enough_observations: nvars >= 1
           ensure
                 correct_observations: number_of_observations = nobs
                 correct_variables: number_of_variables = nvars
                 all_observations_selected: number_of_observations = number_of_selected_observations
                 all_variables_selected: number_of_variables = number_of_selected_variables
                 not_weighted: not weighted
                 weights_set: weights /= void

     make_from_matrix (x: BASIC_MATRIX)
                 -- Create data set based on matrix x.
                 -- Note: x is remembered by reference.
           require
                 x_exists: x /= void
                 x_real: x.real
                 enough_variables: x.width >= 1
                 enough_observations: x.height >= 1
           ensure
                 data_set: data = x
                 correct_observations: number_of_observations = x.height
                 correct_variables: number_of_variables = x.width
                 all_observations_selected: number_of_observations = number_of_selected_observations
                 all_variables_selected: number_of_variables = number_of_selected_variables
                 not_weighted: not weighted
                 weights_set: weights /= void
     
feature -- Access

     data: BASIC_MATRIX
                 -- The entire data set as a matrix

     item (i, j: INTEGER): DOUBLE
                 -- The i'th observation on the j'th variable
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
                 j_large_enough: j >= 1
                 j_small_enough: j <= number_of_variables
           ensure
                 correct_result: Result = data.real_grid_item (i, j)

     selected_data: like Current
                 -- A data structure containing the selected data, weights, etc.
           ensure
                 result_exists: Result /= void
                 weighted_like_x: weighted = Result.weighted
                 correct_number_of_variables: Result.number_of_variables = number_of_selected_variables
                 correct_number_of_observations: Result.number_of_observations = number_of_selected_observations
                 no_copy_if_all_selected: all_selected implies (Result = Current)
     
feature -- Measurement

     Default_format_decimals: INTEGER is 3
                 -- Default number of decimals used by out in formatting output

     Default_format_width: INTEGER is 12
                 -- Default width used by out in formatting output.

     number_of_observations: INTEGER
                 -- The number of observations
           ensure
                 positive_result: Result >= 0

     number_of_selected_observations: INTEGER
                 -- Number of observations currently selected

     number_of_selected_variables: INTEGER
                 -- Number of variables currently selected

     number_of_variables: INTEGER
                 -- The number of variables
           ensure
                 positive_result: Result >= 0
     
feature -- Status report

     all_selected: BOOLEAN
                 -- Is all the data selected?
           ensure
                 Result implies ((number_of_variables = number_of_selected_variables) and (number_of_observations = number_of_selected_observations))

     effective_weights: BASIC_VECTOR
                 -- Weights vector or a vector of all 1.0s if unweighted
           ensure
                 result_exists: Result /= void
                 result_count: Result.count = number_of_observations
                 result_real: Result.real
                 by_reference_if_possible: weighted implies (Result = weights)

     format: FORMAT_SCIENTIFIC
                 -- Used by out to format the individual cells.
                 -- Shared by all instances of this class.
                 -- Change it with format.make (width, decimals)
           ensure
                 result_not_void: Result /= void

     is_selected_observation (i: INTEGER): BOOLEAN
                 -- True if the i'th observation is selected
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
           ensure
                 definition: Result = selected_observations.item (i)

     is_selected_variable (i: INTEGER): BOOLEAN
                 -- True if the i'th variable is selected
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_variables
           ensure
                 definition: Result = selected_variables.item (i)

     selected_observation_numbers: ARRAY [INTEGER]
                 -- Index numbers of the selected observations
           ensure
                 result_exists: Result /= void
                 result_count: Result.count = number_of_selected_observations
                 result_1_based: Result.lower = 1

     selected_observations: ARRAY [BOOLEAN]
                 -- Flags showing which observations are selected

     selected_variable_numbers: ARRAY [INTEGER]
                 -- Index numbers of the selected variables
           ensure
                 result_exists: Result /= void
                 result_count: Result.count = number_of_selected_variables
                 result_1_based: Result.lower = 1

     selected_variables: ARRAY [BOOLEAN]
                 -- Flags showing which variables are selected.

     weight (i: INTEGER): DOUBLE
                 -- Weight of the i_th observation, or 1.0 if unweighted
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
           ensure
                 result_non_negative: Result >= 0.0
                 one_if_not_weighted: not weighted implies Result = 1.0
                 definition: weighted implies Result = weights.real_item (i)

     weighted: BOOLEAN
                 -- Are the weights currently being used?

     weights: BASIC_VECTOR
                 -- Fundamental weight set for the observations
     
feature -- Status setting

     deselect_observation (i: INTEGER)
                 -- Remove observation i from analysis
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
                 observations_left: number_of_selected_observations > 1 or (not is_selected_observation (i))
           ensure
                 i_th_observation_deselected: not is_selected_observation (i)

     deselect_variable (i: INTEGER)
                 -- Remove variable i from analysis
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_variables
                 variable_left: number_of_selected_variables > 1 or (not is_selected_variable (i))
           ensure
                 i_th_variable_deselected: not is_selected_variable (i)

     do_not_use_weights
                 -- Do not use weights in calculation
                 -- Note that setting any weight undoes this, so do it after weights set.
           ensure
                 not_weighted: not weighted

     select_all
                 -- Select all variables and observations
           ensure
                 all_selected: all_selected

     select_observation (i: INTEGER)
                 -- Select observation i for analysis
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
           ensure
                 i_th_observation_selected: is_selected_observation (i)

     select_variable (i: INTEGER)
                 -- Select variable i for analysis
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_variables
           ensure
                 i_th_variable_selected: is_selected_variable (i)

     set_weight (w: DOUBLE; i: INTEGER)
                 -- Set the i th weight to w.
                 -- Note that this sets weighted.
                 -- Variable is deselected if w is 0.0.
                 -- Deselecting an observation does not change its weight, however.
                 -- This makes it easy to select the observation again.
           require
                 i_large_enough: i >= 1
                 i_small_enough: i <= number_of_observations
                 w_large_enough: w >= 0.0
           ensure
                 weight_set: weights.real_item (i) = w
                 weighted: weighted
                 deselected_if_zero: w = 0.0 implies not is_selected_observation (i)

     use_weights
                 -- Use weights in calculation
           ensure
                 weighted: weighted
     
feature -- Element change

     put (value: DOUBLE; i, j: INTEGER)
                 -- Set the i'th observation on the j'th variable to value.
     
feature -- Conversion

     out: STRING
                 -- Output selected data in nice rows, showing effective weights.
     
feature -- Basic operations

     select_from (a: ARRAY [DOUBLE]): ARRAY [DOUBLE]
                 -- Return elements corresponding to selected observations
           require
                 a_exists: a /= void
                 correct_count: a.count = number_of_observations
           ensure
                 result_exists: Result /= void
                 count_right: Result.count = number_of_selected_observations
                 by_reference_if_possible: all_selected implies Result = a
     
invariant

     number_of_selected_variables_large_enough: number_of_selected_variables >= 1
     number_of_selected_observations_large_enough: number_of_selected_observations >= 1
     number_of_selected_variables_small_enough: number_of_selected_variables <= number_of_variables
     number_of_selected_observations_small_enough: number_of_selected_observations <= number_of_observations
     data_exists: data /= void
     data_real: data.real
     data_width: data.width = number_of_variables
     data_height: data.height = number_of_observations
     selected_variables_exists: selected_variables /= void
     selected_variables_count: selected_variables.count = number_of_variables
     selected_observations_exists: selected_observations /= void
     selected_observations_count: selected_observations.count = number_of_observations
     weights_real: weights.real
     weights_count: weights.count = number_of_observations
           -- from ANY
     reflexive_equality: standard_is_equal (Current)
     reflexive_conformance: conforms_to (Current)

end -- class SELECTABLE_DATA
