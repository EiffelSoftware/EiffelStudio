class interface
     INVERTER

create

     make (matrix: BASIC_MATRIX)
                 -- Adopt quadratic matrix as system matrix
                 -- and factorize (triangulize) it.
                 --
                 -- Interface to Nag library procedures:
                 -- f03ahc - non-symmetric complex matrix.
                 -- using A := PLU decomposition.
                 -- f01bnc - symmetric, positive_definite complex matrix,
                 -- using A := U(h)U Cholesky decomposition; 'matrix' elements
                 -- above diagonal need not be set.
                 -- f03afc - non-symmetric real matrix,
                 -- using A := PLU decomposition.
                 -- f03aec - symmetric, positive_definite real matrix,
                 -- using A := LL(t) Cholesky decomposition; 'matrix' elements
                 -- below diagonal need not be set.
                 --
                 -- NOTE:
                 -- (h) stands for conjugate_transposed, (t) - for transposed
                 -- P is a permutation matrix which assures numerical
                 -- stability of the decomposition process.
           require
                 matrix_not_void: matrix /= void
                 matrix_quadratic: matrix.quadratic
                 matrix_real_or_blended_complex: matrix.real or matrix.blended_complex
           ensure
                 matrix_adopted: system = matrix
                 factors_not_void: factors /= void
                 reciprocals_not_void: reciprocals /= void
                 pivots_not_void: pivots /= void
                 positivity_fixed: original_positivity = system.positive_definite
                 reciprocals_used_for_positive_definite_systems: system.positive_definite implies reciprocals_used
                 pivots_used_otherwise: not system.positive_definite implies pivots_used

feature -- Access

     factors: BASIC_MATRIX
                 -- Nag factorized (triangulized) system matrix
                 -- For detailed interpretation of results see
                 -- the corresponding Nag Library functions
                 -- as outlined in 'make'.
                 -- Note that the diagonal elements are implied
                 -- as being all one's or they are kept in
                 -- 'reciprocals' - depending on a decomposition
                 -- method used.

     pivots: ARRAY [INTEGER]
                 -- Row indicies of the pivots
                 -- (Used by non positive definite systems only)

     power_det: INTEGER
                 -- See 'scale_det'

     reciprocals: BASIC_VECTOR
                 -- The reciprocals of the real
                 -- diagonal elements of matrix U
                 -- (Used by positive definite systems only)

     scale_det: COMPLEX
                 -- Scale factor to evaluate the determinant
                 -- as used in expression: scale_det * 2^power_det.
                 -- The separation of the two is to prevent
                 -- a possible overflow.
                 -- For details see Nag Library function
                 -- f03afc (real case) or f03ahc (complex case).
                 --
                 -- Scale_det and power_det are just 'side effects'
                 -- of invoking the 'make'. Note that they are not
                 -- computed by Nag library in the case of
                 -- complex positive definite systems; if you really
                 -- need them for such systems - pretend that
                 -- they are not positive definite.

     system: BASIC_MATRIX
                 -- Quadratic matrix subject to factorization
     
feature -- Status report

     nag_failed: BOOLEAN
                 -- Did the last call to a NAG routine fail?
                 -- (from NAG_EXCEPTIONS)

     nag_ignoring_errors: BOOLEAN
                 -- Will nag_check raise exception if nag_failed?
                 -- (from NAG_EXCEPTIONS)

     nag_printing: BOOLEAN
                 -- Will error messages be printed by NAG?
                 -- (from NAG_EXCEPTIONS)

     nag_recovery: BOOLEAN
                 -- Will NAG return control to us?
                 -- (from NAG_EXCEPTIONS)

     original_positivity: BOOLEAN
                 -- If true - system was positive_definite
                 -- during creation of this object. This
                 -- attribute should be invariant.

     pivots_used: BOOLEAN
                 -- If true the array of pivots was created
                 -- and used during the decomposition process
                 -- in order to assure the numerical
                 -- stability of the process.
                 --
                 -- Pivots can be thus used to create
                 -- a permutation matrix P appearing
                 -- in the factorization of system matrix A:
                 -- A := PLU,
                 -- where L and U are lower and upper
                 -- triangular factors, correspondingly.

     reciprocals_used: BOOLEAN
                 -- True for positive-definite systems.
                 -- Mutually exclusive with 'pivots_used'.
           ensure
                 result_not_pivots_used: Result = not pivots_used
     
feature -- Status setting

     nag_disable_error_printing
                 -- Disable printing of error messages by NAG.
                 -- The error message will be in the argument to raise if recovering.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_will_not_print_errors: not nag_printing

     nag_disable_error_recovery
                 -- Tell NAG to abort if there is an error.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 will_abort: not nag_recovery

     nag_do_not_ignore_errors
                 -- nag_check will raise exception if nag_failed.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 not_nag_ignoring_errors: not nag_ignoring_errors

     nag_enable_error_printing
                 -- Set NAG to print an error message itself if an error occurs.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_will_print_errors: nag_printing

     nag_enable_error_recovery
                 -- Tell NAG to return in case of an error.
                 -- This is the default.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 will_return: nag_recovery

     nag_ignore_errors
                 -- nag_check will not raise exception.
                 -- (from NAG_EXCEPTIONS)
           ensure -- from NAG_EXCEPTIONS
                 nag_ignoring_errors: nag_ignoring_errors

     to_nag: POINTER
                 -- An instance of C type NagError, for passing to NAG routines.

feature -- Basic operations

     inverse: BASIC_MATRIX
                 -- Inverse of the system matrix
                 -- obtained by solving A * X = I
                 -- (I is unit_matrix, one)
           ensure
                 result_exists: Result /= void
                 result_same_kind: Result.same_kind (system)
                 result_quadratic: Result.quadratic
                 result_same_height: Result.height = system.height

     nag_check
                 -- Check result of calling a NAG routine.
                 -- Raise a user exception if it failed.
                 -- (from NAG_EXCEPTIONS)

     nag_error_history: NAG_ERROR_HISTORY
                 -- History of error messages from NAG.
                 -- (from NAG_EXCEPTIONS)

     solution (vectors: BASIC_MATRIX): like vectors
                 -- Solution X of equation A*X = Y,
                 -- where X (Result) and Y (vectors) are matrices
                 -- which columns represent a set of vectors.
                 -- A particular case is an one-column matrices X and Y,
                 -- or the column vectors.
                 --
                 -- Interface to Nag library functions:
                 -- f04akc - general complex system
                 -- f04awc - symmetric, positive definite, complex system
                 -- f04ajc - general real system
                 -- f04agc - symmetric, positive definite, real system
           require
                 vectors_exists: vectors /= void
                 good_height: vectors.height = system.height
                 matching_type: vectors.same_kind (system)
           ensure
                 result_exists: Result /= void
                 result_same_kind: Result.same_kind (system)
                 good_result_height: Result.height = vectors.height
                 good_result_width: Result.width = vectors.width

           require -- from NAG_EXCEPTIONS
                 nag_failed

     nag_last_error_message: STRING
                 -- Error message from failed NAG call.
                 -- (from NAG_EXCEPTIONS)
           require -- from NAG_EXCEPTIONS
                 nag_failed
     
invariant

     factors_not_void: factors /= void
     pivots_not_void: pivots /= void
     positivity_consistent: system.positive_definite = original_positivity
     not_pivots_and_reciprocals: not (pivots_used and reciprocals_used)
