	back is
			-- Move cursor one position backward.
			-- (from RECURSIVE_CURSOR_TREE)
		do
			if below then
				after := False
				before := True
			elseif after then
				after := False
			elseif isfirst then
				before := True
			else

				active := active_parent.child
			end
		end

			-- Move cursor to next position in breadth-first order.
			-- If the active node is the last in
			-- breadth-first order, the cursor ends up off.
			-- (from CURSOR_TREE)

			not_off: not off

	down (i: INTEGER) is
			-- Move cursor one level downward:
			-- to i-th child if there is one,
			-- or after if i = arity + 1,
			-- or before if i = 0.
			-- (from RECURSIVE_CURSOR_TREE)
		require -- from HIERARCHICAL
			not_off: not off
			argument_within_bounds: i >= 1 and i <= arity
		require else -- from CURSOR_TREE
			not_before: not before
			not_after: not after
			not_below: not below
			valid_cursor_index: (above and i = 0) or else valid_cursor_index (i)
		do
			if i = 0 then
				if arity = 0 then
					below := True
				else
					active_parent := active
					active.child_go_i_th (1)
					active := active.child
				end
				before := True
			elseif above or else i <= arity then
				active_parent := active;
				active.child_go_i_th (i);
				active := active.child
			else
				if arity = 0 then
					below := True
				else
					active_parent := active
					active.child_go_i_th (arity)
					active := active.child
				end
				after := True
			end
		ensure then -- from CURSOR_TREE
			gone_before: (i = 0) implies before
		end

	forth is
			-- Move cursor one position forward.
			-- (from RECURSIVE_CURSOR_TREE)
		do
			if below then
				before := False
				after := True
			elseif before then
				before := False

			-- Move cursor to previous position of current level.
			-- (from CURSOR_TREE)

			if not isfirst then

			-- Move cursor to next position of current level.
			-- (from CURSOR_TREE)

					level_forth

			end
		end

			-- Move cursor to next position in postorder.
			-- If the active node is the last in
			-- postorder, the cursor ends up off.
			-- (from CURSOR_TREE)

			not_off: not off

			-- of the l-th level counting from root.
			-- (from CURSOR_TREE)
		require -- from CURSOR_TREE
			argument_within_bounds: l >= 1 and then depth >= l
		do

			-- or above if is_root holds.
			-- (from RECURSIVE_CURSOR_TREE)

			not_off: not off
		require else -- from CURSOR_TREE
			not_above: not above
		do
			if below then
				below := False
			else

				if active /= void then
					active_parent := active.parent
				end
				corresponding_child
			end
			after := False
			before := False
		ensure then -- from CURSOR_TREE
			not_before: not before
			not_after: not after
			not_below: not below
			coherency: (not old off and above) = (old is_root)
		end

			-- onto active node of current tree.

			end
		end

	merge_right (other: CURSOR_TREE [G]) is
			-- Merge the items of other into current structure to
			-- the right of cursor position.
			-- (from CURSOR_TREE)
		require -- from CURSOR_TREE
			other_exists: other /= void
			not_after: not after
			not_above: not above
			only_one_root: (level = 1) implies is_empty
		local
			pos: CURSOR
		do
			if not other.is_empty then
				pos := other.cursor
				other.start
				put_right (other.item)
				forth

					other.down (0)

						other.islast

						other.forth

					up

			end
		end

			-- (from CURSOR_TREE)

			not_before: not before
			not_above: not above

			-- Add v to the right of cursor position.

			not_after: not after
			not_above: not above
			only_one_root: (level = 1) implies is_empty
		do

				active.child_put_right (v)
				active.child_forth
				active_parent := active
